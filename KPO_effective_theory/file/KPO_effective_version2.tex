\begin{align}
    \hat{H} &= \hat{H}_0 + \hat{H}^{\prime}\\[10pt]

    = p(\ket{0}\bra{2} + \ket{4}\bra{2} + \ket{2}\bra{0} + \ket{2}\bra{4})

    \begin{array}{ccc}
    0 & p & 0 \\[5pt]
    p & \Delta & p \\[5pt]
    0 & p & 0
    \end{array}
    \right)
\end{align}
